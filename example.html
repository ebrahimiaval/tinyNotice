OPTIONS

statusTitle : "status title"

statusText : "status sub text"

default tinyNotice statusTitle and statusText [default] {statusTitle:"",statusText:""}

only statusTitle {statusTitle:"your title"}

only statusText {statusText:"your notice text"}

define statusTitle and statusText {statusTitle:"your title" , statusText:"your notice text"}



status : "note || success || warning || error || info"

tinyNotice success status [default] {status : "note"}

tinyNotice success status {status : "success"}

tinyNotice warning status {status : "warning"}

tinyNotice error status {status : "error"}

tinyNotice info status {status : "info"}



lifeTime : 0 || (number per ms)

close [event]

tinyNotice with default lifeTime [default] {lifeTime : 1000}

tinyNotice with 1 second lifeTime {lifeTime : 1000}

tinyNotice with unlimited lifeTime {lifeTime : 0}

close tinyNotice $.tinyNotice("close")
